-- PostgreSQL schema (DDL) - created for document model
CREATE TABLE releases (
  id SERIAL PRIMARY KEY,
  name TEXT NOT NULL,
  version TEXT,
  author TEXT,
  description TEXT,
  release_date DATE,
  extraction_period TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE job_references (
  id SERIAL PRIMARY KEY,
  release_id INTEGER REFERENCES releases(id) ON DELETE CASCADE,
  job_name TEXT NOT NULL,
  job_path TEXT,
  notes TEXT
);

CREATE TABLE dependencies (
  id SERIAL PRIMARY KEY,
  release_id INTEGER REFERENCES releases(id) ON DELETE CASCADE,
  package_name TEXT NOT NULL,
  vendor TEXT,
  version TEXT,
  notes TEXT
);

CREATE TABLE change_history (
  id SERIAL PRIMARY KEY,
  release_id INTEGER REFERENCES releases(id) ON DELETE CASCADE,
  change_date DATE,
  version TEXT,
  responsible TEXT,
  description TEXT
);

CREATE TABLE defects (
  id SERIAL PRIMARY KEY,
  release_id INTEGER REFERENCES releases(id) ON DELETE CASCADE,
  title TEXT,
  description TEXT,
  status TEXT,
  discovered_at DATE
);

-- flexible place to store installation steps/config
CREATE TABLE installation_steps (
  id SERIAL PRIMARY KEY,
  release_id INTEGER REFERENCES releases(id) ON DELETE CASCADE,
  step_order INTEGER,
  instruction TEXT
);

-- example index
CREATE INDEX idx_releases_version ON releases(version);
